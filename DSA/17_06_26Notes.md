### Q1] Longest Palindromic substring
Given a string s, return the longest palindromic substring in s.
#### Input Format:
    "babad"
#### Output Format:
    "bab"
#### Constraints:
    1 <= s.length <= 1000
    s consist of only digits and English letters.
#### Code:
```C++
    #include <bits/stdc++.h>
    using namespace std;

    bool solve(string intended,string typed){
        int l = 0;
        int r = 0;
        while (l < intended.size() && r < typed.size()){
            if (intended[l] == typed[r]){
                l++;
                r++;
            }
            else if (r > 0 && typed[r] == typed[r-1]){
                r++;
            }
            else{
                return false;
            }
        }
        if(l==intended.size() && r == typed.size()){
            return true;
        }
        else{
            while(typed[r] == intended[l-1]){
                r++;
            }
            if (r==typed.size())return true;
            else return false;
        }
    }

        string intended;
        cin >> intended;
        string typed;
        cin >> typed;
```

### Q4] Overlapping Meeting Slots Finder
A scheduling assistant needs to find all common available time slots between two employees, Alice and Bob, for booking meetings. Each employee provides a list of available time slots as closed intervals [start, end] representing time in minutes from midnight. Both lists are pairwise disjoint (no two intervals in the same list overlap) and sorted in ascending order. A common available slot exists when both employees are free at the same time. Your task is to return a list of all intersecting intervals — the time periods when both Alice and Bob are simultaneously available. A closed interval [a, b] includes all real numbers x such that a <= x <= b. The intersection of [a, b] and [c, d] is [max(a,c), min(b,d)] if max(a,c) <= min(b,d), otherwise empty.
#### Input Format:
    3
    2 7
    9 14
    18 25
    3
    3 6
    10 16
    20 23
#### Output Format:
    3 6
    10 14
    20 23
#### Constraints:
- 0 <= p, q <= 1000
- p + q >= 1
- 0 <= start(i) <= end(i) <= 10^9
- 0 <= start(j) <= end(j) <= 10^9
- Intervals within each list are pairwise disjoint and sorted in ascending order.
#### Algorithm:
    step 1: set l = 0, r = 0
    step 2: Repeat until l<len(Alice) and r<len(Bob)
        step 2.1: x = max(Alice[l].first,bob[r].first)
        step 2.2: y = min(Alice[l].second,Bob[r].second)
        step 2.3: if x<= y:
            step 2.3.1: append([x,y] to res)
        step 2.4: if Alice[l].second < Bob[r].second
            step 2.4.1: Increment l
        step 2.5: Else increment r
    step 3: Return result

#### Code:
```C++
#include <iostream>
#include <vector>
using namespace std;

vector<pair<int,int>> findIntersections(vector<pair<int,int>>& alice, vector<pair<int,int>>& bob) {
    //===== Write Your Logic Here =====
    vector<pair<int,int>> ans;
    int l = 0;
    int r = 0;
    while(l<alice.size() && r < bob.size()){
        int x = max(alice[l].first,bob[r].first);
        int y = min(alice[l].second,bob[r].second);
        if (x<= y){
            ans.push_back(make_pair(x,y));
        }
        if (alice[l].second < bob[r].second){
            l++;
        }
        else{
            r++;
        }
    }   
    return ans;

}

    }
```

Given an array nums with n objects colored red, white, or blue, sort them in-place so that objects of the same color are adjacent, with the colors in the order red, white, and blue.

We will use the integers 0, 1, and 2 to represent the color red, white, and blue, respectively.

You must solve this problem without using the library's sort function.
#### Input Format:
    [2,0,2,1,1,0]
#### Output Format:
    [0,0,1,1,2,2]
#### Constraints:
- n == nums.length
- 1 <= n <= 300
- nums[i] is either 0, 1, or 2.
#### Algorithm:
    **Inplace Replacement of the packages**
    step 1: l=0,m=0,r=n-1
    step 2: Repeat until m<=r:
        step 2.1: if packages[m] == 1:
            increment m
        step 2.2: if packages[m] == 0:
            swap packages[l],packages[m] and increment l,m
        step 2.3: else
            swap packages[m],packages[r] and decrement r
    step 3: return packages
#### Code:
```C++

                packages[l] = packages[m];

                m++;
            } else {
                int temp = packages[m];
                packages[m] = packages[r];
                packages[r] = temp;
                r--;
            }
        }
    }
```

### Q1] Peak Average Window
Given an integer array nums of n elements and an integer k, find a contiguous subarray of length exactly k that has the maximum average value and return that average. Any answer with a calculation error less than 10^-5 will be accepted
#### Input Format:
    6
    1 12 -5 -6 50 3
    4
#### Output Format:
    12.75000
#### Constraints:
- 1 <= k <= n <= 10^5
- -10^4 <= nums[i] <= 10^4
#### Algorithm:
    Step 1: left = 0,run_sum = 0;
    Step 2: Perform the given operation on the first k elements and add the running sum in run_sum
    Step 3: right in repeated from k to the last index
    Step 4: include r index and exclude the l index
        Step 4.1:l++
    step 5: Return (double) mx/k;
#### Code:
```C++
#include <iostream>
#include <vector>
#include <iomanip>
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    double peakAverageWindow(vector<int>& nums, int k) {
        int n = nums.size();
        int run_sum = 0;
        for(int i=0;i<k;i++){
            run_sum += nums[i];
        }
        int l = 0;
        int mx = run_sum;
        for(int i=k;i<n;i++){
            run_sum += nums[i];
            run_sum -= nums[l];
            l++;
            mx = max(run_sum,mx);
        }
        return (double)mx/k;
    }
};

int main() {
    int n, k;
    cin >> n;
    vector<int> nums(n);
    for (int i = 0; i < n; i++) cin >> nums[i];
    cin >> k;
    Solution sol;
    cout << fixed << setprecision(5) << sol.peakAverageWindow(nums, k) << "\n";
    return 0;
}
```

### Q4] Nearby Repeated Value
Given an integer array nums and an integer k, return true if there exist two distinct indices i and j in the array such that nums[i] == nums[j] and the absolute difference between i and j is at most k. Otherwise, return false
#### Input Format:
    4
    1 2 3 1
    3
#### Output Format:
    true
#### Constraints:
- 1 <= nums.length <= 10^5
- -10^9 <= nums[i] <= 10^9
- 0 <= k <= 10^5
#### Code:
```C++
#include <iostream>
#include <vector>
#include <unordered_map>
using namespace std;

class Solution {
public:
    bool hasNearbyRepeat(vector<int>& nums, int k) {
        // Write your code here
        unordered_map<int,int> hmap;
        int n = nums.size();
        for(int i=0;i<n;i++){
            if (hmap[nums[i]]){
                int idx = hmap[nums[i]];
                idx--;
                if (abs(idx-i) <= k){
                    return true;
                }
                else{
                    hmap[nums[i]] = i+1;
                }
            }
            else{
                hmap[nums[i]] = i+1;

        return false;
    }
```

#### Consrtaints:
- 1<= arr.size,x <= 10^5
- 0<= arr[i] <= 10^4
#### Code:
```C++
#include <bits/stdc++.h>
using namespace std;

int smallestSubarray(int x, int n, vector<int>& arr) {
    int run_sum = 0;
    int l = 0;
    int mxLen = 100000000;
    int r = 0;
    while(r<n){
        run_sum += arr[r];
        if (run_sum > x){
            while(run_sum > x){
                mxLen = min(mxLen,r-l+1);
                run_sum -= arr[l];
                l++;
            }
        }
        r++;
    }
    if (mxLen == 100000000) return 0;
    else return mxLen;
}
```
