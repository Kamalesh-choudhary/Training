### Q1] Contiguous Subarray
Given a binary array nums, return the maximum length of a contiguous subarray with an equal number of 0 and 1.
#### Input Format:
    [0,1]
#### Output Format:
    2
#### Constraints: 
- 1 <= nums.length <= 105
- nums[i] is either 0 or 1.
#### Algorithm:
    Step 1: Take summ = 0,ans = 0,unordered_map<int,int> hmap
    Step 2: hmap[0] = -1
    Step 3: repeat until 0 to n
        if arr[i] == 0 decrement summ
        if arr[i] == 1 increment summ
        if summ exists in hmap
            ans = max(ans,i-hmap[summ])
        else
            hmap[summ] = i
    Step 4: return ans
#### Code:
```C++
#include <iostream>
#include <vector>
#include <unordered_map>
using namespace std;

class Solution {
public:
    int longestBalanced(vector<int>& arr) {
        int n = arr.size();
        unordered_map<int,int>hmap;
        int summ = 0;   
        int ans = 0;
        hmap[0] = -1;
        for(int i=0;i<n;i++){
            if (arr[i] == 0) summ -= 1;
            else summ += 1;
            if (hmap.find(summ) != hmap.end()) ans = max(ans,i-hmap[summ]);
            else hmap[summ] = i;
        }
        return ans;
    }
};

int main() {
    int n;
    cin >> n;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) cin >> arr[i];
    Solution sol;
    cout << sol.longestBalanced(arr) << endl;
    return 0;
}
```

    [1,1,2,1,1], k = 3

### Q4] Contiguous subarray sum
Given an integer array nums and an integer k, return true if nums has a good subarray or false otherwise.

A good subarray is a subarray where:

    its length is at least two, and
    the sum of the elements of the subarray is a multiple of k.

Note that:

    A subarray is a contiguous part of the array.
    An integer x is a multiple of k if there exists an integer n such that x = n * k. 0 is always a multiple of k.
#### Input Format:
    nums = [23,2,4,6,7], k = 6
#### Output Format:
    true
#### Constraints:
- 1 <= nums.length <= 105
- 0 <= nums[i] <= 109
- 0 <= sum(nums[i]) <= 231 - 1
- 1 <= k <= 231 - 1
#### Code:
```C++
class Solution {
public:
    bool checkSubarraySum(vector<int>& nums, int k) {
        int n = nums.size();
        unordered_map<int,int> hmap;
        hmap[0]=-1;
        int pref = 0;
        for(int i=0;i<n;i++){
            pref += nums[i];
            int rem = pref%k;
            if (hmap.find(rem) != hmap.end()){
                if (i-hmap[rem] >=2) return true;
            }
            else hmap[rem] = i;
        }
        return false;
    }
```

```C++
vector<vector<long long>> buildPrefix(int m, int n, vector<vector<int>>& matrix) {
    vector<vector<long long>> prefix(m,vector<long long>(n,0));
    prefix[0][0] = matrix[0][0];
    for(int i=1;i<n;i++){
        prefix[0][i] = matrix[0][i] + prefix[0][i-1];
    }
    for(int i=1;i<m;i++){
        prefix[i][0] = matrix[i][0] + prefix[i-1][0];

            prefix[i][j] = prefix[i-1][j] + prefix[i][j-1] + matrix[i][j] -prefix[i-1][j-1];
```

### Q13] NeighbourHood Load Aggregation Grid
Ritika is processing a metrics grid where each cell stores the load recorded for one service location. For every cell, she needs the total load across all valid positions within k rows and k columns of that cell. Return the resulting matrix of neighborhood sums.
#### Input Format:
    3 4 1
    2 1 3 4
    5 6 2 1
    7 3 8 9
#### Output Format:
    14 19 17 10
    24 37 37 27
    21 31 29 20
#### Constraints:
- 1 <= m, n <= 100
- 0 <= k <= 100
- 1 <= mat[i][j] <= 100
#### Algorithm:
    Step 1: Build prefix sum matrix for the given matrix
    Step 2: Take ans, prefix = buildPrefix(m,n,mat)
    Step 3: repeat until 0 to m and 0 to n
        row1 = max(0,i-k)
        col1 = max(0,j-k)
        row2 = min(m-1,i+k)
        col2 = min(n-1,j+k)
        tmp = prefix[row2][col2]
        if (row1 > 0) tmp -= prefix[row1-1][col2]
        if (col1 > 0) tmp -= prefix[row2][col1-1]
        if (row1>0 && col1> 0) tmp += prefix[row1-1][col1-1]
        ans[i][j] = tmp
    Step 4: return ans

    buildprefix(m,n,mat):
        Step 1: ans[m][n]
        Step 2: ans[0][0] = mat[0][0]
        Step 3: for i=1 to n
            ans[0][i] = mat[0][i] + ans[0][i-1]
        Step 4: for i=1 to m
            ans[i][0] = mat[i][0] + ans[i-1][0]
        Step 5: for i=1 to m
            for j=1 to n
                ans[i][j] = mat[i][j] + ans[i-1][j] + ans[i][j-1] - ans[i-1][j-1]
        Step 6: return ans
#### Code:
```C++
#include <bits/stdc++.h>
using namespace std;

vector<vector<long long>> buildPrefix(int m, int n, vector<vector<int>>& mat) {
    //===== Write Your Logic Here =====
    vector<vector<long long >> ans(m,vector<long long >(n,0));
    ans[0][0] = mat[0][0];
    for(int i=1;i<n;i++){
        ans[0][i] = mat[0][i] + ans[0][i-1];
    }
    for(int i=1;i<m;i++){
        ans[i][0] = mat[i][0] + ans[i-1][0];
    }
    for(int i=1;i<m;i++){
        for(int j=1;j<n;j++){
            ans[i][j] = mat[i][j] + ans[i-1][j] + ans[i][j-1] - ans[i-1][j-1];
        }
    }
    return ans;
}

vector<vector<long long>> neighborhoodSums(int m, int n, int k, vector<vector<int>>& mat) {
    //===== Write Your Logic Here =====
    vector<vector<long long >> ans;
    vector<vector<long long >> prefix = buildPrefix(m,n,mat);
    for(int i=0;i<m;i++){
        vector<long long> temp;
        for(int j=0;j<n;j++){
            int row1 = max(0,i-k);
            int col1 = max(0,j-k);
            int row2 = min(m-1,i+k);
            int col2 = min(n-1,j+k);
            long long tmp = prefix[row2][col2];
            if (row1 > 0) tmp -= prefix[row1-1][col2];
            if (col1 > 0) tmp -= prefix[row2][col1-1];
            if (row1>0 && col1> 0) tmp += prefix[row1-1][col1-1];
            temp.push_back(tmp);
        }
        ans.push_back(temp);
    }
    return ans;

}

}
```
