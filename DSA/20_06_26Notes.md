### Q1] Create a Linked List
You are given an integer N representing the number of nodes, followed by N integer values. Create a singly linked list by inserting the values in the same order.
#### Input Format:
    5
    10 20 30 40 50
#### Output Format:
    10 20 30 40 50
#### Constraints:
- 1 ≤ N ≤ 10^4
- -10^5 ≤ Node value ≤ 10^5
#### Code:
```C++
#include <bits/stdc++.h>
using namespace std;

class Node{
public:
    int val;
    Node* next;
    Node(int val){
        this->val = val;
        this->next = nullptr;
    }
};

void ins(Node* head,int x){
    Node* temp = head;
    while(temp->next != nullptr){
        temp = temp->next;
    }
    Node* tail = new Node(x);
    temp->next = tail;
}

void display(Node* head){
    Node* temp = head;
    while(temp->next != nullptr){
        cout << temp->val << " " ;
        temp = temp->next;
    }
    cout << temp->val << endl;
}

int main(){
    int n;
    cin >> n;
    Node* head = nullptr;
    for(int i=0;i<n;i++){
        int x;
        cin >> x;
        if (i==0)head = new Node(x);
        else{
            ins(head,x);
        }
    }
    display(head);
}
```

### Q15] Matrix Block sum
Given a m x n matrix mat and an integer k, return a matrix answer where each answer[i][j] is the sum of all elements mat[r][c] for:

    i - k <= r <= i + k,
    j - k <= c <= j + k, and
    (r, c) is a valid position in the matrix.

#### Input Format:
    mat = [[1,2,3],[4,5,6],[7,8,9]], k = 1
#### Output Format:
    [[12,21,16],[27,45,33],[24,39,28]]
#### Constraints:
    m == mat.length
    n == mat[i].length
    1 <= m, n, k <= 100
    1 <= mat[i][j] <= 100

#### Code:
```C++
class Solution {
public:
    vector<vector<int>> buildPrefix(int m, int n,vector<vector<int>>& mat) {
        vector<vector<int>> ans(m, vector<int>(n, 0));
        ans[0][0] = mat[0][0];
        for (int i = 1; i < n; i++) {
            ans[0][i] = mat[0][i] + ans[0][i - 1];
        }
        for (int i = 1; i < m; i++) {
            ans[i][0] = mat[i][0] + ans[i - 1][0];
        }
        for (int i = 1; i < m; i++) {
            for (int j = 1; j < n; j++) {
                ans[i][j] = mat[i][j] + ans[i - 1][j] + ans[i][j - 1] -ans[i - 1][j - 1];
            }
        }
        return ans;
    }
    vector<vector<int>> matrixBlockSum(vector<vector<int>>& mat, int k) {
        vector<vector<int>> ans;
        int m = mat.size();
        int n = mat[0].size();
        vector<vector<int>> prefix = buildPrefix(m, n, mat);
        for (int i = 0; i < m; i++) {
            vector<int> temp;
            for (int j = 0; j < n; j++) {
                int row1 = max(0, i - k);
                int col1 = max(0, j - k);
                int row2 = min(m - 1, i + k);
                int col2 = min(n - 1, j + k);
                long long tmp = prefix[row2][col2];
                if (row1 > 0)
                    tmp -= prefix[row1 - 1][col2];
                if (col1 > 0)
                    tmp -= prefix[row2][col1 - 1];
                if (row1 > 0 && col1 > 0)
                    tmp += prefix[row1 - 1][col1 - 1];
                temp.push_back(tmp);
            }
            ans.push_back(temp);
        }
        return ans;
    }
};
```

### Q16] RangeSum query 2D - Immutable
Given a 2D matrix matrix, handle multiple queries of the following type:

    Calculate the sum of the elements of matrix inside the rectangle defined by its upper left corner (row1, col1) and lower right corner (row2, col2).

Implement the NumMatrix class:

    NumMatrix(int[][] matrix) Initializes the object with the integer matrix matrix.
    int sumRegion(int row1, int col1, int row2, int col2) Returns the sum of the elements of matrix inside the rectangle defined by its upper left corner (row1, col1) and lower right corner (row2, col2).

You must design an algorithm where sumRegion works on O(1) time complexity.
#### Input Format:
    ["NumMatrix", "sumRegion", "sumRegion", "sumRegion"]
    [[[[3, 0, 1, 4, 2], [5, 6, 3, 2, 1], [1, 2, 0, 1, 5], [4, 1, 0, 1, 7], [1, 0, 3, 0, 5]]], [2, 1, 4, 3], [1, 1, 2, 2], [1, 2, 2, 4]]
#### Output Format: 
    [null, 8, 11, 12]
#### Constraints:   
    m == matrix.length
    n == matrix[i].length
    1 <= m, n <= 200
    -104 <= matrix[i][j] <= 104
    0 <= row1 <= row2 < m
    0 <= col1 <= col2 < n
    At most 104 calls will be made to sumRegion.
#### Code:
```C++
class NumMatrix {
public:
    vector<vector<int>> prefix;
    NumMatrix(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        prefix.push_back({matrix[0][0]});
        for(int i=1;i<n;i++){
            int ele = matrix[0][i]+prefix[0][i-1];
            prefix[0].push_back(ele);
        }
        for(int i=1;i<m;i++){
            int ele = matrix[i][0] + prefix[i-1][0];
            prefix.push_back({ele});
        }
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                prefix[i].push_back(matrix[i][j] + prefix[i-1][j] + prefix[i][j-1] - prefix[i-1][j-1]);
            }
        }
    }
    
    int sumRegion(int row1, int col1, int row2, int col2) {
        int ans = prefix[row2][col2];
        if (row1 > 0){
            ans -= prefix[row1-1][col2];
        }
        if (col1 > 0){
            ans -= prefix[row2][col1-1];
        }
        if (row1 > 0 && col1 > 0){
            ans += prefix[row1-1][col1-1];
        }
        return ans;
    }
};
 ```

### Q18] Minimum Lights to illuminate the Road
You are given an integer array lights of length n, representing positions 0 through n - 1 on a road.

For each position i:

    If lights[i] = v, where v > 0, there is a working bulb at position i that illuminates every position from max(0, i - v) to min(n - 1, i + v), inclusive.
    If lights[i] = 0, there is no working bulb at position i.

A position is visible if it is illuminated by at least one working bulb.

You may install additional bulbs at any positions. Each additional bulb installed at position j illuminates positions from max(0, j - 1) to min(n - 1, j + 1), inclusive.

Return the minimum number of additional bulbs required to make every position on the road visible.
#### Input Format:  
    lights = [0,0,0,0]
#### Output Format:
    2
#### Constraints:

    1 <= n == lights.length <= 105
    0 <= lights[i] <= n

 #### Code:
 ```C++
 class Solution {
public:
    int minLights(vector<int>& lights) {
        int n = lights.size();
        vector<int> diff(n+1,0);
        for(int i=0;i<n;i++){
            if (lights[i]>0){
                int v = lights[i];
                int left = max(0,i-v);
                int right = min(n-1,i+v);
                diff[left]++;
                diff[right+1]--;
            }
        }
        int bulbs = 0;
        int gap = 0;
        int run_sum = 0;
        for(int i=0;i<n;i++){
            run_sum += diff[i];
            if (run_sum == 0){
                gap++;
            }
            else{
                if (gap>0){
                    bulbs += (gap+2)/3;
                    gap = 0;
                }
            }
        }
        if (gap > 0){
            bulbs += (gap+2)/3;
        }
        return bulbs;
    }
};
```

### Q19] Finish Time of Tasks I
You are given an integer n representing the number of tasks in a project, numbered from 0 to n - 1. These tasks are connected as a tree rooted at task 0. This is represented by a 2D integer array edges of length n - 1, where edges[i] = [ui, vi] indicates that task ui is the parent of task vi.

You are also given an array baseTime of length n, where baseTime[i] represents the time to complete task i.

The finish time of each task is calculated as follows:

    Leaf task: The finish time is baseTime[i].
    Non-leaf task:
        Let earliest be the minimum finish time among its children, and latest be the maximum finish time among its children.
        Let ownDuration be (latest - earliest) + baseTime[i].
        The finish time of task i is latest + ownDuration.

Return the finish time of the root task 0.
#### Input Format:  
    n = 3, edges = [[0,1],[1,2]], baseTime = [9,5,3]
#### Output Format:
    17
#### Constraints:
    1 <= n <= 105
    edges.length = n - 1
    edges[i] == [ui, vi]
    0 <= ui, vi <= n - 1
    ui != vi
    The input is generated such that edges represents a valid tree.
    baseTime.length == n
    1 <= baseTime[i] <= 105​​​​​​​
#### Code:
```C++
class Solution {
public:
    vector<vector<int>> children;
    vector<int> bt;
    long long dfs(int node){
        if (children[node].empty()){
            return bt[node];
        }
        long long earliest = LLONG_MAX;
        long long latest = LLONG_MIN;
        for(int child:children[node]){
            long long finish = dfs(child);
            earliest = min(earliest,finish);
            latest = max(latest,finish);
        }
        long long ownduaration = (latest-earliest)+bt[node];
        return latest+ownduaration;
    }
    long long finishTime(int n, vector<vector<int>>& edges, vector<int>& baseTime) {
        children.resize(n);
        bt = baseTime;
        for(auto &e:edges){
            children[e[0]].push_back(e[1]);
        }
        return dfs(0);
    }
};
```
