### Q1] Count Distinct Elements in every Window
Given an array arr and an integer k find the count of distinct elements in every window of size k in the array
#### Input Format:
    7
    1 2 1 3 4 2 3 
    4
#### Output Format:
    3 4 4 3
#### Constraints:
- 1<=k<=arr.size<=10^5
- 1<=arr[i]<=10^5
#### Algorithm:
    step 1: Take empty ans array , unordered_map<int,int> hmap,int l=0
    step 2: for i from 0 to k add all elements to the hmap
    step 3: Add the len of hmap to ans
    step 4: Now for i from k to n-1
        step 4.1: add the nums[i] to hmap;
        step 4.2: decrement the nums[l] from hmap
        step 4.3: add len of hmap to ans
        step 4.4: l++
    step 5: return ans
#### Code: 
```C++
#include <iostream>
#include <vector>
#include <unordered_map>
using namespace std;

class Solution {
public:
    vector<int> countDistinct(vector<int>& arr, int k) {
        vector<int> ans;
        int n = arr.size();
        unordered_map<int,int> hmap;
        for(int i=0;i<k;i++){
            hmap[arr[i]]++;
        }
        ans.push_back(hmap.size());
        int l = 0;
        for(int i=k;i<n;i++){
            hmap[arr[i]]++;
            hmap[arr[l]]--;
            if(hmap[arr[l]] == 0){
                hmap.erase(arr[l]);
            }
            ans.push_back(hmap.size());
            l++;
        }
        return ans;
    }
};

int main() {
    int n;
    cin >> n;
    vector<int> arr(n);
    for (int i = 0; i < n; i++) cin >> arr[i];
    int k;
    cin >> k;
    Solution sol;
    vector<int> res = sol.countDistinct(arr, k);
    for (int i = 0; i < (int)res.size(); i++) {
        cout << res[i];
        if (i < (int)res.size() - 1) cout << " ";
    }
    cout << endl;
    return 0;
}
```

### Q2] Shortest canteen Queue That hits the Budget
Given an array of n positive numbers and a positive target . find the minimum length of the subarray Whose sum exceeds or is equal to target.
#### Input Format: 
    8 6
    3 1 2 5 4 2
#### Output Format: 
    2
#### Constraints: 
    1<=TARGET<=10^9
    1<=N<=10^5
    1<=nums[i]<=10^4
#### Algorithm:
    step 1: take ans = INT_MAX
    step 2: take l=0,r=0,run_sum=0
    step 3: Repeat until r<n
        step 3.1: add nums[i] to run_sum
        step 3.2: while run_sum >= target:
            step 3.2.1: ans = min(ans,r-l+1)
            step 3.2.2: run_sum -= nums[l]
            step 3.2.3: l++
        step 3.3: r++
    step 4: return 0 if ans==INT_MAX else ans
#### Code:
```C++
#include <bits/stdc++.h>
using namespace std;

int minSubarrayLen(int target, int n, vector<int>& nums) {
    int run_sum = 0;
    int l = 0;
    int r = 0;
    int ans = INT_MAX;
    while(r<n){
        run_sum += nums[r];
        while (run_sum >= target){
            ans = min(ans,r-l+1);
            run_sum -= nums[l];
            l++;
        }
        r++;
    }
    if (ans == INT_MAX)return 0;
    else return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int target, n;
    cin >> target >> n;
    vector<int> nums(n);
    for (int i = 0; i < n; i++) cin >> nums[i];

    cout << minSubarrayLen(target, n, nums) << "\n";

    return 0;
}
```

### Q3] Minimum window substring
Given two strings s and t of lengths n and m respectively, return the minimum window substring of s such that every character in t (including duplicates) is included in the window.

If there is no such substring, return an empty string "".

If multiple valid windows exist, return the one with the minimum length
#### Input Format:
    ADOBECODEBANC
    ABC
#### Output Format:
    BANC
#### Constraints:
- 1 ≤ s.length ≤ 10^5
- 1 ≤ t.length ≤ 10^5
- `s` and `t` consist of uppercase and lowercase English letters.
- Characters in `t` may appear multiple times.
#### Algorithm:
    step 1: Take an empty string op="" and left = 0;
    step 2: create tewo maps d and e
    step 3: e will be having the number of occurence of all character of t
    step 4: repeat right from0 to len(s):
        step 4.1: Add s[right] tnto map d
        step 4.2: Until d>=e then
            step 4.2.1: get the minimum string (if op is empty or len(op) > right-len+1)
                step 4.2.1.1: op = string from left to right(inclusive of right index)
            step 4.2.2: decrement d of s[left]
            step 4.2.3: if d[s[left]]==0 then remove s[left] from d
            step 4.2.4: increment left
    step 5: return op
#### Code:
```C++
#include <bits/stdc++.h>
using namespace std;

bool exists(unordered_map<char,int>& target,unordered_map<char,int>& curr){
    for(auto [key,val]:target){
        if (curr[key] < val){
            return false;
        }
    }
    return true;
}
```

### Q4] Anagram Pattern Detection in Log Stream
A cybersecurity system monitors a continuous log stream represented as a string logStream for suspicious patterns. A pattern is considered a threat signature if any rearrangement (anagram) of the keyword key appears as a contiguous substring within the log stream. Given a keyword key and a log stream logStream, return the total count of positions in logStream where an anagram of key starts. Note: Two strings are anagrams of each other if one can be formed by rearranging the characters of the other.
#### Input Format:
    forxxorfxdofr
    for
#### Output Format:
    3
#### Constraints:
- 1 <= |key| <= |logStream| <= 100000
- Both strings consist of only lowercase English letters.
#### Algorithm:
    step1: initialise two unordered_maps for target,curr,cnt=0,l=0
    step 2: add the first key.size() of elements in both target and curr
    step 3: for i in range(k,n)
        step 3.1: if are_equal(target,curr)cnt++
        step 3.2: add the ith element in curr
        step 3.3: remove the lth element from curr and handle the curr[key] == 0 issue
        step 3.4: l++
    step 4: if are_equal(target,curr)cnt++ //This is for the handling of the last element added check
    step 5: return cnt
#### Code:
```C++
#include <iostream>
#include <string>
#include <vector>
#include <bits/stdc++.h>
using namespace std;

bool are_equal(unordered_map<char,int>& target,unordered_map<char,int>& curr){
    if (target.size() == curr.size()){
        for(auto [key,val]:target){
            if (curr[key] == val){
                continue;
            }
            else{
                return false;
            }
        }
    }
    else{
        return false;
    }
    return true;
}

int countAnagramOccurrences(string logStream, string key) {
    int k = key.size();
    unordered_map<char,int> target;
    unordered_map<char,int> curr;
    for(int i=0;i<k;i++){
        target[key[i]]++;
        curr[logStream[i]]++;
    }
    int cnt = 0;
    int l = 0;
    int n = logStream.size();
    for(int i=k;i<n;i++){
        if (are_equal(target,curr)){
            cnt++;
        }
        curr[logStream[i]]++;
        curr[logStream[l]]--;
        if (curr[logStream[l]] == 0)curr.erase(logStream[l]);
        l++;
    }
    if (are_equal(target,curr))cnt++;
    return cnt;
}

int main() {
    string logStream, key;
    cin >> logStream >> key;
    cout << countAnagramOccurrences(logStream, key) << endl;
    return 0;
}
```

### Q5] Count subarray sum equals target
Given an array of integers nums and an integer k, return the total number of contiguous subarrays whose elements sum exactly to k
#### Input Format:
    3
    1 1 1
    2
#### Output Format:
    2
#### Constraints:
- 1 <= nums.length <= 2 * 10^4
- -1000 <= nums[i] <= 1000
- -10^7 <= k <= 10^7
#### Code:
```C++
#include <iostream>
#include <vector>
#include <unordered_map>
using namespace std;

class Solution {
public:
    int countTargetSegments(vector<int>& nums, int k) {
        unordered_map<int,int> hmap;
        int prefix = 0,ans = 0;
        hmap[0] = 1;
        for(int x:nums){
            prefix += x;
            if (hmap.count(prefix-k)){
                ans+= hmap[prefix-k];
            }
            hmap[prefix]++;
        }
        return ans;
    }
};

int main() {
    int n;
    long long k;
    cin >> n;
    vector<int> nums(n);
    for (int i = 0; i < n; i++) cin >> nums[i];
    cin >> k;

    Solution sol;
    cout << sol.countTargetSegments(nums, (int)k) << "\n";
    return 0;
}
```

### Q6] Longest Collision Free burst in Autonomous Swarm Telemetry
An autonomous drone swarm operates over a surveillance grid, continuously transmitting telemetry packets to a central ground station. Each packet carries a Flight Signature ID (FSID) — a hardware-stamped identifier assigned to the originating drone unit. The ground station's signal processor buffers incoming packets strictly in the order they arrive. A contiguous segment of buffered packets is declared a Clean Reception Window (CRW) if and only if no FSID appears more than once within that segment. A repeated FSID within an active window indicates a signal collision — the same drone re-transmitted before its earlier packet was acknowledged and evicted from the processor's active register. Given the full telemetry log of N packets in arrival order, determine the length of the longest possible Clean Reception Window. This scalar is used downstream to calibrate the processor's adaptive collision-detection threshold.
#### Input Format:

#### Output Format:
    4
#### Constraints:
- 1 <= N <= 10^6

### Q7] Binary subarrays with sum
Given a binary array nums and an integer goal, return the number of non-empty subarrays with a sum goal.

A subarray is a contiguous part of the array.
#### Input Format:
    nums = [0,0,0,0,0], goal = 0
#### Output Format:
    15
#### Constraints:
- 1 <= nums.length <= 3 * 104
- nums[i] is either 0 or 1.
- 0 <= goal <= nums.length
#### Algorithm:
    step 1: Define a function atmost(k)
        step 1.1: pref = 0,l=0,r=0
        step 1.2: repeat r from 0 to len(arr)
            step 1.2.1: add arr[r] to pref
            step 1.2.2: until pref>k then
                step 1.2.2.1: substrack arr[l] from pref
                step 1.2.2.2: l++
            step 1.2.3: add r-l+1 to ans
        step 1.3: return ans
    step 2: return atmost(k) - atmost(k-1)
#### Code:
```C++
#include <bits/stdc++.h>
using namespace std;

int atmost(vector<int> nums,int k){
    int pref = 0,l=0,r=0,ans=0;
    int n = nums.size();
    while(r<n){
        pref += nums[r];
        while(pref > k){
            pref -= nums[l];
            l++;
        }
        ans += r-l+1;
        r++;
    }
    return ans;
}

long long countSubarrays(int n, vector<int>& arr, int target) {
    return (long long)atmost(arr,target)-atmost(arr,target-1);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;

    vector<int> arr(n);

    for (int i = 0; i < n; i++)
        cin >> arr[i];

    int target;
    cin >> target;

    cout << countSubarrays(n, arr, target) << "\n";

    return 0;
}
```
